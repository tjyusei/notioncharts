<!DOCTYPE html>

<html>

<head>

    <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

</head>

<body>

    <canvas id="myChart"></canvas>


    <script>

        const ctx = document.getElementById('myChart').getContext('2d');


        // 各データのリンク先

        const pageLinks = {

            '本年度の収入': 'https://notion.so/page1', // ← Notionの本年度の収入ページURL

            '前年からの繰越金': 'https://notion.so/page2' // ← Notionの前年からの繰越金ページURL

        };


        // データ

        const data = {

            labels: Object.keys(pageLinks),

            datasets: [{

                data: [22375, 2675], // 各カテゴリの数値

                backgroundColor: ['#F0B90B', '#4682B4'] // 色指定（黄色、青）

            }]

        };


        // チャートの作成

        const myChart = new Chart(ctx, {

            type: 'doughnut',

            data: data,

            options: {

                onClick: function(event, elements) {

                    if (elements.length > 0) {

                        const index = elements[0].index;

                        const label = data.labels[index];

                        const url = pageLinks[label];


                        if (url) {

                            window.open(url, '_blank'); // 新しいタブで開く

                        }

                    }

                }

            }

        });

    </script>

</body>

</html>
